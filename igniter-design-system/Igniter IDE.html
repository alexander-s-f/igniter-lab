-- settlement.ig — history-aware ledger
module SparkCRM.Ledger
import Lang.Core.Money

contract Settlement {
  input   account_id: String
  input   base_bid:   Decimal[2]
  input   tax_rate:   Decimal[4]

  read    ledger: History[Decimal[2]] from TBackend lifecycle: :durable

  compute gross_bid = mul(base_bid, tax_rate)
  compute prior     = ledger.as_of(clock.now)
  compute delta     = gross_bid - prior

  invariant non_negative predicate: gross_bid >= 0.00 severity: :error

  output  gross_bid: Decimal[2]
  output  delta:     Decimal[2]
